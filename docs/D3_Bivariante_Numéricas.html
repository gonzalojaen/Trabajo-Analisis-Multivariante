StudentPerformanceFactors <- read.csv("StudentPerformanceFactors_with_Comments.csv")
variables_numericas <- sapply(StudentPerformanceFactors, is.numeric)
datos_numericos <- StudentPerformanceFactors[, variables_numericas]

matriz_correlaciones <- cor(datos_numericos, use = "complete.obs")

corrplot(matriz_correlaciones, method = "circle", type = "upper", 
         col = colorRampPalette(c("#f7f7f7", "#2166ac"))(200),
         tl.cex = 0.8, tl.col = "black", cl.cex = 0.8)

pares_alta_correlacion <- which(abs(matriz_correlaciones) > 0.3 & abs(matriz_correlaciones) < 1, arr.ind = TRUE)
pares_alta_correlacion <- unique(t(apply(pares_alta_correlacion, 1, sort))) # Evitar duplicados

for (i in 1:nrow(pares_alta_correlacion)) {
  var1 <- colnames(datos_numericos)[pares_alta_correlacion[i, 1]]
  var2 <- colnames(datos_numericos)[pares_alta_correlacion[i, 2]]
  
  print(ggplot(datos_numericos, aes_string(x = var1, y = var2)) +
          geom_point(alpha = 0.4, color = "#4c72b0") +
          geom_smooth(method = "lm", color = "#d55e00", fill = "#d55e00", alpha = 0.2) +
          theme_minimal(base_size = 14) +
          labs(title = paste("Relaci√≥n entre", var1, "y", var2),
               x = var1, y = var2) +
          theme(plot.title = element_text(hjust = 0.5)))
}
