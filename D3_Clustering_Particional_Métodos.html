# Leer archivo CSV
datos <- read.csv("StudentPerformanceFactors_1.csv")
tipos <- sapply(datos, class)

varNum <- names(tipos)[which(tipos %in% c("integer", "numeric"))]

Aplicar K-means

set.seed(123)  # Para reproducibilidad
k <- 4  #  Numero √≥ptimo de clusters
kmeans_res <- kmeans(datos[, varNum], centers = k, nstart = 25)

# Visualizar los clusters
fviz_cluster(kmeans_res, data = datos[, varNum], geom = "point",
             ellipse.type = "norm", main = "Clusters con K-means (k=4)")

Aplicar PAM (Partitioning Around Medoids)

pam_res <- pam(datos[, varNum], k = 4)

# Visualizar PAM
fviz_cluster(pam_res, geom = "point", ellipse.type = "norm", main = "Clusters con PAM (k=4)")

Aplicar CLARA (Clustering Large Applications)

set.seed(123)
clara_res <- clara(datos[, varNum], k = 4, samples = 5)  # 5 muestras aleatorias

# Visualizar los clusters
fviz_cluster(clara_res, geom = "point", ellipse.type = "norm", main = "Clusters con CLARA (k=4)")

Comparar con Silhouette Score

# Silhouette Score para K-Means
sil_kmeans <- silhouette(kmeans_res$cluster, dist(datos[, varNum]))
mean_sil_k_means <- mean(sil_kmeans[, 3])  # Promedio del coeficiente de silueta

# Silhouette Score para PAM
sil_pam <- silhouette(pam_res$clustering, dist(datos[, varNum]))
mean_sil_PAM <- mean(sil_pam[, 3])  # Promedio del coeficiente de silueta

# Coeficiente de silueta para CLARA
sil_clara <- silhouette(clara_res$clustering, dist(datos[, varNum]))
mean_sil_CLARA <- mean(sil_clara[, 3])  # Promedio del coeficiente de silueta

# Data frame con los resultados
silhouette_scores <- data.frame(
  Method = c("K-Means", "PAM", "CLARA"),
  Silhouette_Score = c(mean_sil_k_means, mean_sil_PAM, mean_sil_CLARA)
)

print(silhouette_scores)
