Anlisis grfico univariante

Ahora, en la seccin visual, presentaremos tanto los barplots de cada variable como los pie charts para obtener una representacin grfica de los datos.

for (vC in varCat) {
  cat(paste0("================= ", vC, " ================\n"))
  
  # Guardamos la informacin numrica 
  frecAbs <- table(dades[, vC])
  frecRel <- prop.table(frecAbs)
  # Convertir tabla a data.frame con nombres correctos
  datos <- as.data.frame.table(frecAbs)
  colnames(datos) <- c("Var1", "Freq")  # Asegurar nombres correctos
  
  # Barplot 
  g1 <- ggplot(datos, aes(x = Var1, y =Freq, fill = Var1)) +
    geom_bar(stat = "identity") +
    labs(title = "Barplot",
         x = "Categora",
         y = "Frecuencia") +
    theme_minimal() + scale_fill_viridis_d()
  
  # PieChart
  datos$frecuencia_relativa <- datos$Freq / sum(datos$Freq)
  datos$porcentaje <- paste0(round(datos$frecuencia_relativa * 100, 1), "%")
  
  g2 <- ggplot(datos, aes(x = "", y = frecuencia_relativa, fill = Var1)) +
    geom_bar(stat = "identity", width = 1) +
    coord_polar(theta = "y") +
    labs(title = "PieChart") +
    theme_void() +
    geom_text(aes(label = porcentaje), 
              position = position_stack(vjust = 0.5), 
              color = "white", 
              size = 5)
  print(g1+g2)
}
